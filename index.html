from js import document, setInterval from pyodide.ffi import create_proxy import random canvas = document.getElementById("game") ctx = canvas.getContext("2d") TILE = 32 dungeon = [ "##########", "#........#", "#..##....#", "#........#", "##########" ] # player x = TILE * 2 y = TILE * 2 speed = 3 # enemy ex = TILE * 6 ey = TILE * 2 edir = random.choice(["up", "down", "left", "right"]) espeed = 2 keys = set() def key_down(event): keys.add(event.key) def key_up(event): keys.discard(event.key) document.addEventListener("keydown", create_proxy(key_down)) document.addEventListener("keyup", create_proxy(key_up)) def is_wall(px, py): tile_x = int(px // TILE) tile_y = int(py // TILE) if tile_y < 0 or tile_y >= len(dungeon): return True if tile_x < 0 or tile_x >= len(dungeon[0]): return True return dungeon[tile_y][tile_x] == "#" def draw_map(): for row_i, row in enumerate(dungeon): for col_i, tile in enumerate(row): if tile == "#": ctx.fillStyle = "gray" ctx.fillRect(col_i * TILE, row_i * TILE, TILE, TILE) def move_enemy(): global ex, ey, edir nx, ny = ex, ey if edir == "up": ny -= espeed elif edir == "down": ny += espeed elif edir == "left": nx -= espeed elif edir == "right": nx += espeed if is_wall(nx, ny): edir = random.choice(["up", "down", "left", "right"]) else: ex, ey = nx, ny def draw(): global x, y new_x, new_y = x, y if "w" in keys: new_y -= speed if "s" in keys: new_y += speed if "a" in keys: new_x -= speed if "d" in keys: new_x += speed if not is_wall(new_x, new_y): x, y = new_x, new_y move_enemy() ctx.fillStyle = "black" ctx.fillRect(0, 0, canvas.width, canvas.height) draw_map() # enemy ctx.fillStyle = "crimson" ctx.fillRect(ex, ey, 28, 28) # player ctx.fillStyle = "gold" ctx.fillRect(x, y, 28, 28) setInterval(create_proxy(draw), 16)
